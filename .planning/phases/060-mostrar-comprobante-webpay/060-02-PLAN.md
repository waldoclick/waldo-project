2. **Create computed property to transform Order payment_response → IWebpayReceipt:**
```typescript
const receiptData = computed((): IWebpayReceipt | null => {
  if (!data.value || 'error' in data.value) return null;
  
  const paymentResponse = data.value.payment_response as Record<string, unknown> | undefined;
  if (!paymentResponse) return null;

  // Map Webpay API fields to IWebpayReceipt interface
  return {
    amount: data.value.amount as number | undefined,
    authorizationCode: paymentResponse.authorization_code as string | undefined,
    transactionDate: paymentResponse.transaction_date as string | undefined,
    paymentType: paymentResponse.payment_type_code as string | undefined,
    cardNumber: (paymentResponse.card_detail as Record<string, unknown> | undefined)?.card_number as string | undefined,
    buyOrder: data.value.buy_order as string | undefined,
    commerceCode: paymentResponse.commerce_code as string | undefined,
    status: paymentResponse.status as string | undefined,
  };
});
```
