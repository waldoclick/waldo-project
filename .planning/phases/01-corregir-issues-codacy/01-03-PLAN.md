      ```
      function randomInt(max: number): number {
        const limit = Math.floor(0xffffffff / max) * max;
        const buf = new Uint32Array(1);
        let value: number;
        do {
          crypto.getRandomValues(buf);
          value = buf[0] ?? 0;
        } while (value >= limit);
        return value % max;
      }
      ```
