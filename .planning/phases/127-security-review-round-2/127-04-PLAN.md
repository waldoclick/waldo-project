    ```typescript
    import DOMPurify from "isomorphic-dompurify";

      renderer: {
        html() { return ""; }, // block-level raw HTML → suppress
      },
      walkTokens(token) {
        if (token.type === "html") {

        }
      },

    ```

    ```typescript
    const sanitizeHTML = (
      html: string,
      allowedTags: string[] = [],
      allowedAttrs: string[] = [],
    ): string => {
      if (!html) return "";
      return DOMPurify.sanitize(html, {
        ALLOWED_TAGS: allowedTags,
        ALLOWED_ATTR: allowedAttrs,
        KEEP_CONTENT: true,
      });
    };
    ```

    3) Keep `sanitizeText`, `sanitizeRich`, `sanitizeStrict`, `sanitizeBasic` exactly as they are (they delegate to sanitizeHTML with their tag/attr lists) — preserve the return object names so the five components keep working.

    ```typescript
    const parseMarkdown = (markdown: string): string => {
      if (!markdown) return "";
      const html = marked.parse(markdown, { async: false }) as string;
      return sanitizeRich(html);
    };
    ```
