**Step 1 template:**
- Heading: "Buscar noticias"

- Button "Buscar" (`btn btn--primary`) calls `handleSearch()`, disabled while `loading`

**`handleSearch()` method:**
```typescript
async function handleSearch() {
  if (!query.value.trim() || loading.value) return;
  loading.value = true;
  try {
    const result = await $fetch<{ news: TavilyNewsResult[] }>('/api/search/tavily', {
      method: 'POST',
      body: { query: query.value.trim(), num: 10 }
    });
    searchResults.value = result.news || [];
  } catch (e) {
    console.error('[LightBoxArticles] Tavily error:', e);
    searchResults.value = [];
  } finally {
    loading.value = false;
  }
}
```

**`handleSelectArticle(item: TavilyNewsResult)` method:**
```typescript
async function handleSelectArticle(item: TavilyNewsResult) {
  if (loading.value) return;
  loading.value = true;
  try {
    // Fetch article HTML via Strapi proxy (avoids CORS)
    const html = await $fetch<string>(`/api/fetch-url?url=${encodeURIComponent(item.link)}`)
      .catch(() => ''); // HTML fetch is best-effort — don't block step advance
    selectedArticle.value = {
      title: item.title,
      url: item.link,
      date: item.date,
      html: html || '',
    };
    currentStep.value = 2;
  } finally {
    loading.value = false;
  }
}
```

**IMPORTANT NOTE on HTML fetch:** There is no `/api/fetch-url` endpoint yet — use a simpler approach. Instead of fetching article HTML via proxy (which requires a new Strapi endpoint not in scope), just advance to Step 2 with the available metadata (title, url, date) and include `html: ''` in `selectedArticle`. The Gemini prompt will use the URL and title directly. Update `handleSelectArticle` to NOT make a fetch call:

**`handleGenerate()` method:**
```typescript
async function handleGenerate() {
  if (!geminiPrompt.value.trim() || loading.value) return;
  loading.value = true;
  try {
    const result = await $fetch<{ text: string }>('/api/ia/gemini', {
      method: 'POST',
      body: { prompt: geminiPrompt.value }
    });
    // Parse JSON from Gemini text response
    const parsed = JSON.parse(result.text) as {
      title: string;
      header: string;
      body: string;
      keywords: string[];
      source_url: string;
      source_date: string;
    };
    generatedArticle.value = parsed;
    currentStep.value = 3;
  } catch (e) {
    console.error('[LightBoxArticles] Gemini error:', e);
  } finally {
    loading.value = false;
  }
}
```
