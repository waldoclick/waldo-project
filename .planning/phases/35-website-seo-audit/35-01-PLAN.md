```typescript
// plugins/seo.ts
import { defineNuxtPlugin } from "#app";

const BASE_URL = process.env.BASE_URL || "http://localhost:3000";
const DEFAULT_IMAGE = `${BASE_URL}/images/share.jpg`;

export default defineNuxtPlugin((nuxtApp) => {
  nuxtApp.provide(
    "setSEO",
    (params: {
      title: string;
      description: string;
      imageUrl?: string;
      url?: string;
      ogType?: string;
      twitterCard?: string;
    }) => {
      useSeoMeta({
        title: params.title,
        description: params.description,
        ogTitle: params.title,
        ogDescription: params.description,
        ogImage: params.imageUrl || DEFAULT_IMAGE,
        ogUrl: params.url,
        ogType: params.ogType || "website",
        twitterCard: (params.twitterCard || "summary_large_image") as
          | "summary"
          | "summary_large_image"
          | "app"
          | "player",
        twitterTitle: params.title,
        twitterDescription: params.description,
      });
    },
  );
});

declare module "#app" {
  interface NuxtApp {
    $setSEO: (params: {
      title: string;
      description: string;
      imageUrl?: string;
      url?: string;
      ogType?: string;
      twitterCard?: string;
    }) => void;
  }
}
```

Note: The `twitterCard` type cast is needed because `useSeoMeta` expects a specific union type, not `string`. The cast to the union type is safe here because we control the default and the interface restricts callers to valid string values.
