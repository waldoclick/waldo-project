async getUserTotalViews(userId: number): Promise<number> {
  const activeAds = await strapi.db.query("api::ad.ad").findMany({ where: { user: userId, active: true }, select: ["id"] });
  if (activeAds.length === 0) return 0;
  const adIds = activeAds.map(a => a.id);
  return strapi.db.query("api::ad-view.ad-view").count({ where: { ad: { $in: adIds } } });
}
