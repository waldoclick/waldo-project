    **1. Create composable** `apps/dashboard/app/composables/useExportCsv.ts`:

    ```typescript
    import { ref } from "vue";
    import type { Order } from "@/types/order";
    import { ordersTocsv } from "@/utils/csv";

    export function useExportCsv() {
      const apiClient = useApiClient();
      const isExporting = ref(false);

      async function exportOrders(): Promise<void> {
        isExporting.value = true;
        try {
          const res = await apiClient<{ data: Order[] }>("orders", {
            method: "GET",
            params: {
              pagination: { page: 1, pageSize: 999999 },
              sort: "createdAt:desc",
              populate: ["user", "ad"],
            } as unknown as Record<string, unknown>,
          });
          const orders = Array.isArray(res.data) ? res.data : [];
          const csv = ordersTocsv(orders);
          const blob = new Blob([csv], { type: "text/csv;charset=utf-8;" });
          const url = URL.createObjectURL(blob);
          const link = document.createElement("a");
          link.setAttribute("href", url);
          link.setAttribute("download", `orders-${new Date().toISOString().slice(0, 10)}.csv`);
          document.body.appendChild(link);
          link.click();
          document.body.removeChild(link);
          URL.revokeObjectURL(url);
        } finally {
          isExporting.value = false;
        }
      }

      return { exportOrders, isExporting };
    }
    ```

    IMPORTANT: Use the existing dashboard `orders` endpoint with a very large pageSize instead of the Strapi `export-csv` endpoint. This avoids needing to parse raw CSV from the API response — the composable gets typed Order[] objects and uses the already-tested `ordersTocsv()` utility to format them. The Strapi export-csv endpoint remains available for direct API consumers but the dashboard UI path is simpler through the existing find endpoint.

    ALTERNATIVE (if the Strapi export-csv endpoint from Plan 01 is preferred): Replace the apiClient call with a raw fetch to `orders/export-csv` that returns a string, then create the Blob from that string directly without ordersTocsv(). But this approach loses the typing benefit.

    USE THE FIRST APPROACH (fetch JSON, format client-side with ordersTocsv).
