<objective>
Allow Google-only users to create a local password by reusing the existing forgot-password token flow.

Purpose: Users who registered via Google OAuth cannot log in with email/password because their provider='google'. This phase makes `overrideForgotPassword` send a branded "create password" email when it detects a Google-only account, and makes `overrideResetPassword` flip `provider` to `'local'` after the reset succeeds — enabling both login methods without breaking One Tap.

- Modified `overrideForgotPassword` — Google-only detection, conditional template/subject
- Modified `overrideResetPassword` — provider flip post-reset
- New Jest tests covering all 5 requirements
- 2 pre-existing test failures fixed within the describe blocks this plan extends
</objective>

describe("overrideResetPassword", () => {
  beforeEach(() => {
    jest.clearAllMocks();
    // Restore global strapi mock (registerUserLocal's beforeEach overrides it)
    (global as unknown as Record<string, unknown>).strapi = {
      db: { query: strapiQueryMock },
      plugins: {
        "users-permissions": {
          services: { jwt: { issue: mockJwtIssue } },
        },
      },
      log: { error: jest.fn() },
    };
  });

  describe("GOAUTH-128-03: provider flip after successful reset", () => {
    it("flips provider to 'local' when user was Google-only", async () => {
      // Arrange
      const mockReset = jest.fn(async (ctx) => {
        ctx.response.body = { jwt: "new-jwt", user: { id: 42 } };
      });
      mockUserFindOne.mockResolvedValueOnce({ id: 42, provider: "google" });
      mockUserUpdate.mockResolvedValue({ id: 42 });

      const ctx = {
        request: { body: { password: "NewPass1234", code: `${Date.now().toString(16)}:abc` } },
        response: { body: null as unknown },
        body: null as unknown,
        badRequest: jest.fn(),
        send: jest.fn(),
      };

      // Act
      const handler = overrideResetPassword(mockReset);
      await handler(ctx);

      // Assert: original reset was called
      expect(mockReset).toHaveBeenCalled();
      // Assert: DB update called with provider:'local'
      expect(mockUserUpdate).toHaveBeenCalledWith(
        expect.objectContaining({
          where: { id: 42 },
          data: { provider: "local" },
        }),
      );
    });

    it("does NOT update provider when user was already provider:'local'", async () => {
      // Arrange
      const mockReset = jest.fn(async (ctx) => {
        ctx.response.body = { jwt: "new-jwt", user: { id: 99 } };
      });
      mockUserFindOne.mockResolvedValueOnce({ id: 99, provider: "local" });

      const ctx = {
        request: { body: { password: "NewPass1234", code: `${Date.now().toString(16)}:abc` } },
        response: { body: null as unknown },
        body: null as unknown,
        badRequest: jest.fn(),
        send: jest.fn(),
      };

      // Act
      const handler = overrideResetPassword(mockReset);
      await handler(ctx);

      // Assert: no update call made
      expect(mockUserUpdate).not.toHaveBeenCalled();
    });

    it("does NOT attempt provider lookup when response body has no user.id (failed reset)", async () => {
      // Arrange — reset produces no body (failure path)
      const mockReset = jest.fn(async (ctx) => {
        ctx.response.body = null;
      });

      const ctx = {
        request: { body: { password: "NewPass1234", code: `${Date.now().toString(16)}:abc` } },
        response: { body: null as unknown },
        body: null as unknown,
        badRequest: jest.fn(),
        send: jest.fn(),
      };

      // Act
      const handler = overrideResetPassword(mockReset);
      await handler(ctx);

      // Assert: no DB lookup, no update
      expect(mockUserFindOne).not.toHaveBeenCalled();
      expect(mockUserUpdate).not.toHaveBeenCalled();
    });
  });
});
