// Tool definition shape
const tools: Anthropic.Tool[] = [{
  name: "web_search",
  description: "Search the web for current information",
  input_schema: {
    type: "object",
    properties: {
      query: { type: "string", description: "Search query" }
    },
    required: ["query"]
  }
}];
// Create message
const response = await this.client.messages.create({
  model: this.modelName,
  max_tokens: 4096,
  tools,
  messages
});
// response.stop_reason: "end_turn" | "tool_use" | "max_tokens" | "stop_sequence"

// TextBlock: { type: "text", text: string }
// Tool result shape for messages array:
// { role: "user", content: [{ type: "tool_result", tool_use_id: string, content: string }] }

         - Append `{ role: "assistant", content: response.content }` to `messages`

           - Cast `block.input` as `{ query: string }`
           - Call `this.executeBraveSearch(block.input.query)`
           - Append a `tool_result` message:
             ```
             { role: "user", content: [{
               type: "tool_result",
               tool_use_id: block.id,
               content: JSON.stringify(results)
             }]}
             ```
      4. Continue loop

    **`executeBraveSearch()` method:**
    ```typescript
    private async executeBraveSearch(query: string): Promise<WebSearchResult[]> {
      const url = `https://api.search.brave.com/res/v1/web/search?q=${encodeURIComponent(query)}&count=5`;
      const response = await fetch(url, {
        headers: {
          "Accept": "application/json",
          "X-Subscription-Token": this.braveApiKey,
        },
      });
      if (!response.ok) {
        throw new Error(`Brave Search API error: ${response.status} ${response.statusText}`);
      }
      const data = await response.json() as { web?: { results?: WebSearchResult[] } };
      return data.web?.results ?? [];
    }
    ```

      async claude(ctx: Context): Promise<void> {
        const body = ctx.request.body as { prompt?: string };
        const prompt = body?.prompt?.trim();

        if (!prompt) {

          return;

          throw new ApplicationError(`Anthropic API error: ${message}`);
        }
      },
